网页不能直接写入私人 Google Sheet。请把下面 Apps Script 部署为 Web App（执行身份：你；访问权限：Anyone with the link），再把部署 URL 粘贴到这里。

const SPREADSHEET_ID = '1bfIeytT6UMdvWXimeg4s1HVuXHqmpYZx53ufsbes6Ms';
const SHEET_GID = 1256735562;

function doPost(e) {
  const data = e.parameter || JSON.parse(e.postData.contents || '{}');
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheets().find(s => s.getSheetId() === SHEET_GID) || ss.getSheets()[0];
  sheet.appendRow([new Date(), data.flight, data.date, data.serial, data.seat, data.rawText]);
  return ContentService.createTextOutput(JSON.stringify({ ok: true })).setMimeType(ContentService.MimeType.JSON);
}
